# env.lockfile.osx-arm64.txt — subsideo v1.1 (Phase 1 ENV-10 / D-16)
# Two-layer explicit lockfile for bit-for-bit reproducible rebuild on osx-arm64.
#
# Layer 1 (conda): exported via
#     micromamba list -n subsideo --explicit --md5 --no-pip
# Layer 2 (pip):   pip-only packages (not tracked by conda-meta) enumerated by
#     pip list --format=freeze then filtered against conda-meta/*.json names
#
# Rebuild recipe:
#     micromamba create -n subsideo --file env.lockfile.osx-arm64.txt --no-pip-autodetect
#     # (the conda URL section installs the conda layer)
#     pip install --no-deps -r <(awk '/^# --- PIP LAYER ---/,/^$/' env.lockfile.osx-arm64.txt | grep '==')
#
# NOTE: micromamba 2.5.0 has an upstream bug (invalid argument in pip inspect)
# that prevents a single-command `--explicit --md5` export when pip packages are
# present. This two-layer split is the interim workaround; sunset when micromamba
# >=2.6 ships with a fixed pip inspection path.
