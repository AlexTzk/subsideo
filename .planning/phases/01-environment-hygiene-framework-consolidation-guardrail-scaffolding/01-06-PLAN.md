```python
Public helpers (ENV-06 / D-Research-§F):

    download_reference_with_retry   -- per-source-aware HTTPS/S3 download with retry + resume

    credential_preflight            -- fail-fast env var check
    bounds_for_burst                -- wrap opera_utils + subsideo.burst.db fallback (ENV-08)

Per-source RETRY_POLICY dict -- Earthdata 401 must NOT retry forever
(PITFALLS P0.4); CloudFront 403 refreshes URL, not retries.

DESIGN CONSTRAINT: this module MUST NOT import from subsideo.products.* --
it is a peer of validation/compare_*.py (Failure-Mode Boundaries §ARCHITECTURE).
Conda-forge-only libraries (rasterio, isce3, dolphin) are imported inside

# -- Retry policy ---------------------------------------------------------

RetrySource = Literal["CDSE", "EARTHDATA", "CLOUDFRONT", "HTTPS"]

RETRY_POLICY: dict[str, dict[str, Any]] = {
    "CDSE": {
        "retry_on":  [429, 503, "OutOfMemoryError"],
        "abort_on":  [401, 403, 404],
    },
    "EARTHDATA": {
        "retry_on":  [429, 503],
        "abort_on":  [401, 403, 404],
    },
    "CLOUDFRONT": {
        "retry_on":  [503, "ExpiredToken"],
        "abort_on":  [401, 404],
        "refresh_url_on": [403],
    },
    "HTTPS": {
        "retry_on":  [429, 503],
        "abort_on":  [401, 403, 404],

class ReferenceDownloadError(Exception):
    """Raised when download_reference_with_retry hits an abort_on status."""

    def __init__(self, source: str, status: int | str, url: str) -> None:
        super().__init__(
            f"Abort: source={source} status={status} url={url} "
            f"(per RETRY_POLICY[{source!r}]['abort_on'])"
        )
        self.source = source
        self.status = status
        self.url = url


# -- 1. Burst bounds ------------------------------------------------------

def bounds_for_burst(
    burst_id: str,
    buffer_deg: float = 0.2,
) -> tuple[float, float, float, float]:
    """Return (west, south, east, north) in degrees for a burst_id, buffered.

    Primary path: `opera_utils.burst_frame_db.get_burst_id_geojson([burst_id])`.
    Fallback path: `subsideo.burst.db.query_bounds(burst_id)` for EU bursts
    not yet in opera-utils' N.Am.-only DB (RESEARCH.md Open Question 2).

    Mitigates PITFALLS P1.1 cached-bias (hand-coded bounds drift from reality).
    """
    try:
        from opera_utils.burst_frame_db import get_burst_id_geojson
        gdf = get_burst_id_geojson([burst_id], as_geodataframe=True)
        if len(gdf) == 0:
            raise ValueError(f"Burst {burst_id!r} not in opera_utils burst_frame_db")
        west, south, east, north = gdf.total_bounds
    except Exception as first_err:
        logger.debug("opera_utils lookup failed for {}: {}; trying EU burst DB", burst_id, first_err)
        try:
            from subsideo.burst.db import query_bounds  # Task 2 adds this if missing
            west, south, east, north = query_bounds(burst_id)
        except Exception as second_err:
            raise ValueError(
                f"Burst {burst_id!r} not in opera_utils or subsideo EU DB: "
                f"({first_err}; fallback: {second_err})"
            ) from second_err

    return (
        float(west - buffer_deg),
        float(south - buffer_deg),
        float(east + buffer_deg),
        float(north + buffer_deg),
    )

def bounds_for_mgrs_tile(
    tile_id: str,
    buffer_deg: float = 0.1,
) -> tuple[float, float, float, float]:
    """Return (west, south, east, north) in degrees for an MGRS-100km tile, buffered.

    Primary path: `opera_utils.burst_frame_db` MGRS helpers (if available in
    the installed opera-utils version). Fallback path: lookup against the
    `src/subsideo/validation/_mgrs_tiles.geojson` seed file that ships with
    subsideo; the seed covers the v1.1 eval-script tile IDs (10TFK, 29TNF,
    33TXP, ... — every tile used by any run_eval_*.py at v1.1 lands here).

    Used by:
      - run_eval_dswx.py (Lake Balaton tile 33TXP) — Plan 01-07 Task 3
      - Imported but not called by run_eval_dist*.py (dist_s1 auto-derives
        bounds from mgrs_tile_id internally) — Plan 01-07 Task 3

    Raises ValueError if the tile_id is not found in either source.
    """
    # (1) Primary: opera_utils MGRS helpers (if present).
    try:
        from opera_utils.burst_frame_db import get_mgrs_tile_geojson  # noqa: F401
        # Note: the exact function name in opera_utils may differ between
        # releases. If the attribute lookup fails, fall through to the seed
        # file. This is intentionally defensive.
        gdf = get_mgrs_tile_geojson([tile_id], as_geodataframe=True)
        if len(gdf) >= 1:
            west, south, east, north = gdf.total_bounds
            return (
                float(west - buffer_deg),
                float(south - buffer_deg),
                float(east + buffer_deg),
                float(north + buffer_deg),
            )
    except (ImportError, AttributeError) as e:
        logger.debug("opera_utils MGRS lookup unavailable ({}); using seed file", e)

    # (2) Fallback: seed GeoJSON shipped with subsideo
    import json
    from importlib.resources import files as _pkg_files

    seed_path = _pkg_files("subsideo.validation").joinpath("_mgrs_tiles.geojson")
    try:
        data = json.loads(seed_path.read_text())
    except (FileNotFoundError, OSError) as e:
        raise ValueError(
            f"MGRS tile {tile_id!r} not in opera_utils and seed file unreadable: {e}"
        ) from e

    for feat in data.get("features", []):
        props = feat.get("properties", {})
        if props.get("tile_id") == tile_id or props.get("mgrs_tile") == tile_id:
            geom = feat.get("geometry", {})
            if geom.get("type") == "Polygon":
                coords = geom["coordinates"][0]
                lons = [c[0] for c in coords]
                lats = [c[1] for c in coords]
                return (
                    float(min(lons) - buffer_deg),
                    float(min(lats) - buffer_deg),
                    float(max(lons) + buffer_deg),
                    float(max(lats) + buffer_deg),
                )
    raise ValueError(
        f"MGRS tile {tile_id!r} not in opera_utils or "
        f"{seed_path} (seed covers 10TFK, 29TNF, 33TXP, ... — "
        f"add the missing tile if v1.1 eval scripts require it)"
    )


# -- 2. Credential preflight ---------------------------------------------

def credential_preflight(env_vars: Sequence[str]) -> None:
    """Fail fast if any required env var is missing / empty.

    Raises SystemExit with a one-line message listing all missing vars.
    Called at the top of every eval script (replacing ad-hoc `if not os.environ.get(...)` blocks).
    """
    missing = [v for v in env_vars if not os.environ.get(v)]
    if missing:
        raise SystemExit(
            f"credential_preflight: the following env vars are not set or empty: "
            f"{missing}. Check .env or shell export."
        )


# -- 3. OPERA frame selection ---------------------------------------------

def select_opera_frame_by_utc_hour(
    sensing_datetime: datetime,
    frame_metadata: Sequence[dict[str, Any]],
    *,
    tolerance_hours: float = 1.0,
) -> dict[str, Any]:
    """Return the OPERA frame_metadata entry whose sensing UTC hour matches.

    Used when asf-search returns multiple candidate OPERA frames for the same
    burst -- the correct frame is the one whose `sensing_datetime` (exact UTC
    hour + spatial footprint) matches the query (REL-03 methodological
    finding: "OPERA frame selection by exact UTC hour + spatial footprint").

    Raises ValueError if zero or multiple frames match within tolerance_hours.
    """
    target_hour = sensing_datetime.replace(minute=0, second=0, microsecond=0)
    matches: list[dict[str, Any]] = []
    for frame in frame_metadata:
        ft = frame.get("sensing_datetime")
        if ft is None:
            continue
        if isinstance(ft, str):
            ft = datetime.fromisoformat(ft.replace("Z", "+00:00"))
        delta = abs((ft - target_hour).total_seconds())
        if delta <= tolerance_hours * 3600:
            matches.append(frame)
    if not matches:
        raise ValueError(
            f"No OPERA frame within {tolerance_hours}h of {sensing_datetime}"
        )
    if len(matches) > 1:
        raise ValueError(
            f"Multiple OPERA frames within {tolerance_hours}h of {sensing_datetime}: "
            f"{[f.get('id') or f.get('granule') for f in matches]}"
        )
    return matches[0]


# -- 4. Resume safety -----------------------------------------------------

def ensure_resume_safe(cache_dir: Path, manifest_keys: Sequence[str]) -> bool:
    """Return True if cache_dir contains every expected manifest key file.

    Non-destructive: never deletes or truncates files. Returns False if any
    listed key is missing -- caller should either re-download or abort.
    """
    cache_dir = Path(cache_dir)
    if not cache_dir.exists():
        return False
    try:
        existing = {p.name for p in cache_dir.iterdir()}
    except (OSError, PermissionError) as e:
        logger.warning("ensure_resume_safe: cannot read {}: {}", cache_dir, e)
        return False
    return all(k in existing for k in manifest_keys)


# -- 5. Download with per-source retry -----------------------------------

def _coerce_status(resp: requests.Response) -> int:
    try:
        return int(resp.status_code)
    except Exception:
        return -1


def download_reference_with_retry(
    url: str,
    dest: Path,
    *,
    source: RetrySource,
    max_retries: int = 5,
    backoff_seconds: int = 30,
    session: requests.Session | None = None,
) -> Path:
    """Download *url* to *dest* with per-source retry policy.

    Raises
    ------
    ValueError
        If `source` is not a recognised RETRY_POLICY key (explicit source
        required; no default — PITFALLS P0.4 mitigation).
    ReferenceDownloadError
        If the response status is in RETRY_POLICY[source]['abort_on'].
    requests.RequestException
        After max_retries exhausted on retryable statuses.
    """
    if source not in RETRY_POLICY:
        raise ValueError(
            f"download_reference_with_retry: unknown source {source!r} "
            f"(expected one of {list(RETRY_POLICY)})"
        )
    policy = RETRY_POLICY[source]
    sess = session or requests.Session()
    dest = Path(dest)
    dest.parent.mkdir(parents=True, exist_ok=True)

    attempt = 0
    current_url = url
    while attempt < max_retries:
        attempt += 1
        try:
            with sess.get(current_url, stream=True, timeout=300) as resp:
                status = _coerce_status(resp)
                if status in policy.get("abort_on", []):
                    raise ReferenceDownloadError(source, status, current_url)
                if source == "CLOUDFRONT" and status in policy.get("refresh_url_on", []):
                    logger.warning(
                        "CloudFront URL expired (status 403); caller must refresh "
                        "the signed URL -- aborting with ReferenceDownloadError."
                    )
                    raise ReferenceDownloadError(source, status, current_url)
                if status in policy.get("retry_on", []):
                    wait = min(backoff_seconds * (2 ** (attempt - 1)), 300)
                    logger.info(
                        "Retry {}/{} for {} ({}): status={}, sleeping {}s",
                        attempt, max_retries, source, current_url, status, wait,
                    )
                    time.sleep(wait)
                    continue
                resp.raise_for_status()
                # 2xx: stream to dest
                tmp = dest.with_suffix(dest.suffix + ".partial")
                with tmp.open("wb") as f:
                    for chunk in resp.iter_content(chunk_size=1024 * 1024):
                        if chunk:
                            f.write(chunk)
                tmp.replace(dest)
                logger.info("Downloaded {} -> {}", current_url, dest)
                return dest
        except ReferenceDownloadError:
            raise
        except requests.RequestException as e:
            logger.warning(
                "Transport error for {} attempt {}/{}: {}",
                current_url, attempt, max_retries, e,
            )
            time.sleep(min(backoff_seconds * (2 ** (attempt - 1)), 300))
    raise requests.RequestException(
        f"download_reference_with_retry exhausted {max_retries} attempts for {url}"
    )
```
