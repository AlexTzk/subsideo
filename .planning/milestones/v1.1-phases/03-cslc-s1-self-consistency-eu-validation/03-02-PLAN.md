    def count_opera_cslc_granules_2024(bbox: tuple) -> int:
        """Count OPERA CSLC-S1 granules in 2024-01..2025-12 over bbox."""
        try:
            results = earthaccess.search_data(
                short_name="OPERA_L2_CSLC-S1_V1",
                bounding_box=bbox,
                temporal=(datetime(2024, 1, 1), datetime(2025, 12, 31)),
            )
            return len(list(results))
        except Exception as e:  # noqa: BLE001
            logger.warning("earthaccess CSLC query failed for {}: {}", bbox, e)
            return 0


    def derive_dominant_burst_id(bbox: tuple) -> str | None:
        """Return the dominant JPL-lowercase burst_id for the bbox via
        earthaccess -> opera_utils.get_burst_id parsing."""
        try:
            from collections import Counter

            import opera_utils  # lazy
            results = earthaccess.search_data(
                short_name="OPERA_L2_CSLC-S1_V1",
                bounding_box=bbox,
                temporal=(datetime(2024, 1, 1), datetime(2024, 12, 31)),
                count=50,
            )
            names = [r.data_links()[0].split("/")[-1] for r in results[:50]]
            burst_ids = []
            for n in names:
                try:
                    bid = opera_utils.get_burst_id(n).lower()
                    burst_ids.append(bid)
                except Exception:
                    continue
            if not burst_ids:
                return None
            most_common, _ = Counter(burst_ids).most_common(1)[0]
            return most_common
        except Exception as e:  # noqa: BLE001
            logger.warning("Burst-id derivation failed for {}: {}", bbox, e)
            return None


    # Typical EGMS L2a PS density estimates (W10 fix — density selection by explicit
    # `region` parameter, not by stringifying the bbox tuple). All values are
    # back-of-envelope from BOOTSTRAP §2.3 + EGMS 2019_2023 release metadata:
    #   - "EU"  : ~300 PS/km^2 over Iberian semi-arid / Mediterranean plateau
    #   - "NAM" : EGMS does NOT cover North America — caller receives None
    EGMS_PS_DENSITY_PER_KM2: dict[str, float] = {
        "EU": 300.0,   # Iberian-Meseta / Alentejo / Massif Central semi-arid rock
    }


    def count_egms_stable_ps_ceiling(
        bbox: tuple[float, float, float, float],
        region: str,
    ) -> int | None:
        """Rough ceiling estimate of EGMS L2a stable PS in bbox.

        W10 fix: `region` drives the density dict lookup (NOT bbox string-match).
        Callers pass `region="EU"` for Iberian candidates and `region="NAM"` for
        Mojave candidates. NAM returns None because EGMS covers EU only — there
        is no EGMS ceiling to report for N.Am. AOIs.

        Parameters
        ----------
        bbox : (west, south, east, north) in EPSG:4326 degrees.
        region : "EU" | "NAM"

        Returns
        -------
        int | None
            Back-of-envelope count of "stable PS" (approx. area_km2 × density
            × 0.3-stable-fraction). None when region has no EGMS coverage.
        """
        if region not in EGMS_PS_DENSITY_PER_KM2:
            return None
        import math
        west, south, east, north = bbox
        lat_km = (north - south) * 111.0
        lon_km = (east - west) * 111.0 * math.cos(math.radians((north + south) / 2))
        area_km2 = lat_km * lon_km
        density = EGMS_PS_DENSITY_PER_KM2[region]
        ceiling = int(area_km2 * density * 0.3)  # 30% of PS typically stable per L2a
        return ceiling

    def _fifteen_epochs_for_candidate(
        c: CandidateAOI,
    ) -> list[datetime]:
        """Return 15 consecutive S1A 12-day epochs covering the candidate bbox.

        Strategy:
          1. Query asf_search for SENTINEL-1A IW SLC scenes intersecting the bbox
             over a 2024-01-01 through 2024-06-30 window (168-day rolling window;
             covers 15 × 12-day visits).
          2. Filter to POEORB-only epochs where possible (avoid the POEORB->RESORB
             transition ~21 days after acquisition; POEORB is in steady-state by
             end of January for early-January acquisitions).
          3. Pick the first 15 consecutive epochs on the dominant relative orbit.

        On failure (network error, fewer than 15 consecutive scenes), return a
        synthetic 15-entry list of `datetime(2024, m, d, 14, 1, 16)` at 12-day
        cadence starting 2024-01-13, so the artifact always emits 15 entries.
        The implementer MUST flag synthetic fills in the generated markdown so
        the user can override before Plan 03-03/04 lock.
        """
        # Deterministic fallback list matching SoCal window (same orbit family
        # is unrealistic for some Mojave AOIs, but this preserves structural
        # shape; the actual probe run writes ASF-derived epochs when possible).
        fallback = [datetime(2024, m, d, 14, 1, 16) for m, d in [
            (1, 13), (1, 25), (2,  6), (2, 18),
            (3,  1), (3, 13), (3, 25), (4,  6),
            (4, 18), (4, 30), (5, 12), (5, 24),
            (6,  5), (6, 17), (6, 29),
        ]]
        try:
            west, south, east, north = c.bbox
            results = asf.search(
                platform=asf.PLATFORM.SENTINEL1A,
                processingLevel=asf.PRODUCT_TYPE.SLC,
                intersectsWith=f"POLYGON(({west} {south}, {east} {south}, {east} {north}, {west} {north}, {west} {south}))",
                start="2024-01-01T00:00:00Z", end="2024-06-30T23:59:59Z",
            )
            epochs = sorted({r.properties["startTime"] for r in results})
            parsed = [datetime.fromisoformat(e.rstrip("Z")) for e in epochs]
            # Simple de-dup on calendar date (burst-level re-acquisitions may repeat)
            seen_dates: set = set()
            unique: list[datetime] = []
            for p in parsed:
                k = p.date().isoformat()
                if k in seen_dates:
                    continue
                seen_dates.add(k)
                unique.append(p)
            if len(unique) >= 15:
                return unique[:15]
            logger.warning(
                "{}: only {} ASF scenes in window; emitting synthetic fallback list",
                c.aoi, len(unique),
            )
            return fallback
        except Exception as e:  # noqa: BLE001
            logger.warning("{}: ASF 15-epoch query failed ({}); using fallback", c.aoi, e)
            return fallback
