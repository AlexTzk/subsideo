```python
    """Raised when EFFIS_WFS_URL is unset (lock file absent or candidate_chosen=neither)."""


def _validate_lock() -> None:

def _build_retry_session():
    """Return a ``requests.Session`` mounted with a urllib3 Retry adapter
    keyed on ``RETRY_POLICY['EFFIS']`` per the D-18 amendment.

    The session is passed to ``WebFeatureService(...)`` so that owslib's
    underlying HTTP layer applies the same retry semantics that
    ``harness.download_reference_with_retry`` would have applied. The
    harness still owns the policy declaration; this function maps it
    onto the WFS-in-memory dispatch path.
    """
    import requests
    from urllib3.util.retry import Retry

    policy = RETRY_POLICY["EFFIS"]
    # Project the policy onto urllib3's Retry primitives. Numeric entries
    # in retry_on map to status_forcelist; string entries
    # ('ConnectionError', 'TimeoutError') translate to allowed_methods +
    # connect/read retry counts (urllib3 retries on those exception types
    # by default for idempotent methods).
    status_forcelist = [s for s in policy["retry_on"] if isinstance(s, int)]
    abort_on = policy.get("abort_on", [])
    retry = Retry(
        total=5,
        backoff_factor=2,  # 2, 4, 8, 16, 32, 60-cap; matches the harness 60s cap
        status_forcelist=status_forcelist,
        allowed_methods=frozenset(["GET", "POST"]),
        raise_on_status=False,
    )
    sess = requests.Session()
    adapter = requests.adapters.HTTPAdapter(max_retries=retry)
    sess.mount("https://", adapter)
    sess.mount("http://", adapter)
    # Stash the abort_on list on the session so the caller can post-check
    # the final response status against it; urllib3 doesn't itself raise
    # on those (we want a ReferenceDownloadError on 401/403/404).
    sess._effis_abort_on = abort_on  # type: ignore[attr-defined]
    return sess


@dataclass(frozen=True)
class EFFISFetchResult:
    """Lightweight wrapper around the WFS response + query metadata.

    Public consumers receive a tuple ``(gdf, meta)`` rather than this dataclass
    directly -- the dataclass is for internal organisation only.
    """
    gdf: object  # gpd.GeoDataFrame; not type-annotated to keep geopandas lazy
    meta: object  # EFFISQueryMeta; from matrix_schema (lazy import at call site)


def fetch_effis_perimeters(
    event_id: str,
    bbox_wgs84: tuple[float, float, float, float],
    date_start: date,
    date_end: date,
    cache_dir: Path,
) -> tuple["object", "object"]:  # (gpd.GeoDataFrame, EFFISQueryMeta)
    """Fetch EFFIS burnt-area perimeters for a bbox + date range, cache to disk.

    Network access uses ``owslib.wfs.WebFeatureService`` directly with a
    retry-mounted ``requests.Session`` driven by ``RETRY_POLICY['EFFIS']``
    (D-18 amendment in ROADMAP Phase 5 scope-amendment block).

    Parameters
    ----------
    event_id : str
        Identifier used in cache path; must be one of the
        ``DistEUEventID`` Literal values (aveiro / evros / spain_culebra).
    bbox_wgs84 : tuple[float, float, float, float]
        ``(west, south, east, north)`` in EPSG:4326 degrees.
    date_start, date_end : date
        Inclusive date range for the ``firedate`` property filter.
    cache_dir : Path
        Top-level cache directory (e.g., ``Path("eval-dist_eu")``). The
        response is cached at ``cache_dir/event_id/effis_perimeters/perimeters.geojson``.

    Returns
    -------
    tuple[gpd.GeoDataFrame, EFFISQueryMeta]
        The GeoDataFrame in EPSG:4326 + the per-event query metadata for
        meta.json reproducibility audit (CONTEXT D-19).

    Raises
    ------
    EFFISLockMissingError
        If module-level endpoint constants are unset.
    ReferenceDownloadError
        Raised when the WFS response status appears in
        ``RETRY_POLICY['EFFIS']['abort_on']`` (401 / 403 / 404). Mirrors
        the abort semantics of ``harness.download_reference_with_retry``
        even though we don't call that helper.
    """
    _validate_lock()
    cache_path = cache_dir / event_id / "effis_perimeters" / "perimeters.geojson"
    cache_path.parent.mkdir(parents=True, exist_ok=True)

    # Lazy imports
    from io import BytesIO
    import geopandas as gpd
    from owslib.etree import etree
    from owslib.fes import And, BBox, PropertyIsBetween
    from owslib.wfs import WebFeatureService

    # WFS 2.0.0 wraps filters in fes2.Filter; WFS 1.1.0 uses fes1
    if EFFIS_FILTER_NAMESPACE == "fes2":
        from owslib.fes2 import Filter as Filter2
    else:
        from owslib.fes import FilterCapabilities as Filter2  # WFS 1.1.0 fallback

    from subsideo.validation.matrix_schema import EFFISQueryMeta

    # Build the OGC filter
    filter_obj = Filter2(
        And([
            BBox(list(bbox_wgs84), "urn:ogc:def:crs:EPSG::4326"),
            PropertyIsBetween(
                propertyname=EFFIS_DATE_PROPERTY,
                lower=date_start.isoformat(),
                upper=date_end.isoformat(),
            ),
        ])
    )
    filter_xml = etree.tostring(filter_obj.toXML()).decode("utf-8")

    # Cache hit short-circuits (CONTEXT D-19 warm re-run path)
    if cache_path.exists() and cache_path.stat().st_size > 0:
        logger.info("EFFIS cache hit for {} at {}", event_id, cache_path)
        gdf = gpd.read_file(cache_path)
        meta = EFFISQueryMeta(
            wfs_endpoint=EFFIS_WFS_URL,
            layer_name=EFFIS_LAYER_NAME,
            filter_string=filter_xml,
            response_feature_count=len(gdf),
            fetched_at=(
                datetime.fromtimestamp(cache_path.stat().st_mtime, tz=timezone.utc)
                .isoformat()
            ),
        )
        return gdf, meta

    # Cache miss -- live WFS fetch with RETRY_POLICY['EFFIS']-driven retries
    logger.info("EFFIS WFS fetch for {} bbox={} dates=[{}..{}]", event_id, bbox_wgs84, date_start, date_end)
    session = _build_retry_session()
    abort_on = getattr(session, "_effis_abort_on", [401, 403, 404])
    wfs = WebFeatureService(
        url=EFFIS_WFS_URL,
        version="2.0.0",
        timeout=60,
        # Pass the retry-mounted session so owslib uses our urllib3.Retry adapter.
        # Some owslib versions accept `headers` only; if `auth`/`session` is unsupported,
        # the retry adapter still applies via owslib's default requests session if we
        # set it via `wfs.session = session` post-construction.
    )
    try:
        wfs.session = session  # type: ignore[attr-defined]
    except AttributeError:
        # owslib version doesn't expose .session; the retry policy falls back
        # to the harness-declared semantics applied at the eval-script level
        # (per-event try/except in run_eval_dist_eu.py catches transport

        else:
            gdf = gdf.to_crs("EPSG:4326")

        wfs_endpoint=EFFIS_WFS_URL,
        layer_name=EFFIS_LAYER_NAME,
        filter_string=filter_xml,
        response_feature_count=len(gdf),
        fetched_at=datetime.now(timezone.utc).isoformat(),
    )
    return gdf, meta


def rasterise_perimeters_to_grid(
    gdf: "object",  # gpd.GeoDataFrame in target CRS already
    out_shape: tuple[int, int],
    transform: "object",  # rasterio.Affine
) -> tuple[np.ndarray, np.ndarray]:
    """Dual-rasterise EFFIS perimeters to subsideo's 30 m DIST grid (CONTEXT D-17).

    Parameters
    ----------
    gdf : gpd.GeoDataFrame
        Perimeters in the SAME CRS as ``transform`` (caller reprojects to UTM
        before calling). Empty gdf raises ValueError.
    out_shape : tuple[int, int]
        ``(height, width)`` of the target raster in pixels.
    transform : rasterio.Affine
        Affine transform for the target raster.

    Returns
    -------
    mask_at_false : np.ndarray (uint8, shape out_shape)
        ``rasterio.features.rasterize(all_touched=False)`` -- only pixels
        whose centre falls inside a polygon. PRIMARY (gate value) per
        CONTEXT D-17; F1 against this mask is the dist:eu cell metric.
    mask_at_true : np.ndarray (uint8, shape out_shape)
        ``rasterio.features.rasterize(all_touched=True)`` -- pixels touched
        by polygon boundary. DIAGNOSTIC ONLY per CONTEXT D-17; expected
        ~+2-4pp F1 inflation per PITFALLS P4.4.

    Raises
    ------
    ValueError
        If ``gdf`` is empty (no perimeters to rasterise) -- caller is
        responsible for handling zero-feature events.
    """
    _validate_lock()
    if len(gdf) == 0:
        raise ValueError(
            "Cannot rasterise empty GeoDataFrame -- no EFFIS perimeters to project. "
            "Caller should record zero-feature event in metrics.json and skip."
        )

    from rasterio.enums import MergeAlg
    from rasterio.features import rasterize

    shapes = [(geom, 1) for geom in gdf.geometry]

    mask_at_false = rasterize(
        shapes=shapes,
        out_shape=out_shape,
        transform=transform,
        fill=0,
        dtype="uint8",
        all_touched=False,  # CONTEXT D-17 primary
        merge_alg=MergeAlg.replace,
    )
    mask_at_true = rasterize(
        shapes=shapes,
        out_shape=out_shape,
        transform=transform,
        fill=0,
        dtype="uint8",
        all_touched=True,  # CONTEXT D-17 diagnostic
        merge_alg=MergeAlg.replace,
    )
    return mask_at_false, mask_at_true
```
