Use this Python script invoked via `micromamba run -n subsideo python3 -c '<script>'`. The script writes its full output to stdout; you append the stdout to the lock file.

```python
# Probe script (not committed; one-shot probe action).
import sys
import json
from datetime import date, datetime, timezone
from io import BytesIO

CANDIDATES = [
    ("A", "https://maps.effis.emergency.copernicus.eu/effis", "ms:modis.ba.poly"),
    ("B", "http://geohub.jrc.ec.europa.eu/effis/wfs", "EFFIS:BurntAreasAll"),
]

PROBE_DATES = [
    ("aveiro_2024", "2024-09-15", "2024-09-25"),
    ("evros_2023", "2023-08-19", "2023-09-08"),
    ("spain_culebra_2022", "2022-06-15", "2022-06-22"),
]

print(f"# Probe run at: {datetime.now(timezone.utc).isoformat()}")
print()

for label, url, primary_layer in CANDIDATES:
    print(f"## Candidate {label}: {url}")
    print(f"### GetCapabilities")
    try:
        from owslib.wfs import WebFeatureService
        wfs = WebFeatureService(url=url, version="2.0.0", timeout=30)
        contents = list(wfs.contents)
        print(f"GetCapabilities OK -- {len(contents)} layers available")
        # Print first 30 layer names
        for name in contents[:30]:
            print(f"  - {name}")
        if len(contents) > 30:
            print(f"  ... and {len(contents) - 30} more")
        # Check for primary_layer + alternates
        for candidate in [primary_layer, "ms:modis.ba.poly", "EFFIS:BurntAreasAll",
                          "EFFIS:BurntAreas7Days", "EFFIS:BurntAreas30Days",
                          "modis.ba.poly", "BurntAreas"]:
            if candidate in contents:
                print(f"  [layer present] {candidate}")
    except Exception as e:
        print(f"GetCapabilities FAILED: {type(e).__name__}: {e}")
        print()
        continue

    print(f"### GetFeature smoke test (typename={primary_layer})")
    for event_id, start, end in PROBE_DATES:
        try:
            from owslib.fes import And, BBox, PropertyIsBetween
            from owslib.fes2 import Filter as Filter2
            from owslib.etree import etree
            # Use a wide European bbox to maximise hit chance (we are testing
            # whether the layer + filter syntax work, not whether per-event AOIs
            # are correct).
            bbox = (-10.0, 35.0, 30.0, 50.0)
            filter_obj = Filter2(
                And([
                    BBox(list(bbox), "urn:ogc:def:crs:EPSG::4326"),
                    PropertyIsBetween(propertyname="firedate", lower=start, upper=end),
                ])
            )
            filter_xml = etree.tostring(filter_obj.toXML()).decode("utf-8")
            response = wfs.getfeature(typename=primary_layer, filter=filter_xml)
            data = response.read()
            # Try to parse as GML / shapezip via geopandas
            try:
                import geopandas
                gdf = geopandas.read_file(BytesIO(data))
                print(f"  {event_id} ({start}..{end}): {len(gdf)} features (firedate property accepted)")
            except Exception as ge:
                # Geopandas parse failed; report raw response size as a fallback
                print(f"  {event_id} ({start}..{end}): {len(data)} bytes returned (geopandas parse failed: {ge})")
        except Exception as e:
            print(f"  {event_id} ({start}..{end}): GetFeature FAILED: {type(e).__name__}: {e}")
    print()
```
