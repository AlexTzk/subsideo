```python
def test_dist_cmd_iterates_results(tmp_path, mocker):
    """B-06: CLI dist command iterates list[DISTResult], not single result."""
    from typer.testing import CliRunner
    from subsideo.cli import app

    mock_result_ok = MagicMock()
    mock_result_ok.valid = True
    mock_result_ok.validation_errors = []

    mock_result_fail = MagicMock()
    mock_result_fail.valid = False
    mock_result_fail.validation_errors = ["tile failed"]

    # Return a list of results (the contract)
    mocker.patch(
        "subsideo.products.dist.run_dist_from_aoi",
        return_value=[mock_result_ok, mock_result_fail],
    )

    aoi_file = tmp_path / "aoi.geojson"
    aoi_file.write_text('{"type":"Polygon","coordinates":[[[11,48],[12,48],[12,49],[11,49],[11,48]]]}')

    runner = CliRunner()
    result = runner.invoke(app, [
        "dist",
        "--aoi", str(aoi_file),
        "--start", "2025-01-01",
        "--end", "2025-02-01",
        "--out", str(tmp_path / "out"),
    ])
    assert result.exit_code == 1  # one tile failed
    assert "FAIL" in result.output or "FAIL" in (result.stderr or "")
```

Note: Check the actual CLI parameter names in cli.py for --aoi, --start, --end, --out and adjust accordingly. The dist_cmd function uses typer.Option annotations -- match those names exactly.
